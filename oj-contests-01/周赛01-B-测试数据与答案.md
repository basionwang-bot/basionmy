## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <string>
using namespace std;

int cnt[26];

int main() {
    string s;
    cin >> s;
    for (int i = 0; i < 26; i++) cnt[i] = 0;
    for (char c : s)
        cnt[c - 'a']++;
    for (int i = 0; i < 26; i++)
        if (cnt[i] > 0)
            cout << (char)('a' + i) << ":" << cnt[i] << endl;
    return 0;
}
```
