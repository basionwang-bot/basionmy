## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

struct Sailor {
    string name;
    int age, years;
};

bool cmp(const Sailor& a, const Sailor& b) {
    if (a.years != b.years) return a.years > b.years;
    return a.name < b.name;
}

int main() {
    int n;
    cin >> n;
    vector<Sailor> sailors;
    for (int i = 0; i < n; i++) {
        Sailor s;
        cin >> s.name >> s.age >> s.years;
        if (s.age >= 18 && s.years >= 2)
            sailors.push_back(s);
    }
    if (sailors.empty()) {
        cout << "none\n";
    } else {
        sort(sailors.begin(), sailors.end(), cmp);
        for (auto& s : sailors) cout << s.name << "\n";
    }
    return 0;
}
```

**关键点**：先筛选（年龄 ≥ 18 且年限 ≥ 2），再排序（年限降序，同年限按姓名升序）。
