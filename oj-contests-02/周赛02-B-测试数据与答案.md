## 测试点 4（最大最小同在下标 1）

**输入**
```
4
10 20 30 40
```

**预期输出**
```
40 4 10 1
```

---

## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

const int N = 1005;
int a[N];

int main() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> a[i];
    int maxVal = a[1], maxIdx = 1;
    int minVal = a[1], minIdx = 1;
    for (int i = 2; i <= n; i++) {
        if (a[i] > maxVal) { maxVal = a[i]; maxIdx = i; }
        if (a[i] < minVal) { minVal = a[i]; minIdx = i; }
    }
    cout << maxVal << " " << maxIdx << " " << minVal << " " << minIdx << endl;
    return 0;
}
```
