## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

struct Seg {
    int l, r;
} a[1005];

bool cmp(const Seg& x, const Seg& y) {
    return x.r < y.r;
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i].l >> a[i].r;
    sort(a, a + n, cmp);
    int cnt = 1, pos = a[0].r;
    for (int i = 1; i < n; i++) {
        if (a[i].l > pos) {
            cnt++;
            pos = a[i].r;
        }
    }
    cout << cnt << endl;
    return 0;
}
```
