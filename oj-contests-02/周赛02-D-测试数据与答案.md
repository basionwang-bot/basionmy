## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

struct Book {
    string name, author;
    int price, stock;
} b[105];

bool cmp(const Book& x, const Book& y) {
    if (x.price != y.price) return x.price < y.price;
    if (x.stock != y.stock) return x.stock > y.stock;
    return x.name < y.name;
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> b[i].name >> b[i].author >> b[i].price >> b[i].stock;
    sort(b, b + n, cmp);
    for (int i = 0; i < n; i++)
        cout << b[i].name << " " << b[i].author << " " << b[i].price << " " << b[i].stock << endl;
    return 0;
}
```
