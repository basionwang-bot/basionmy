## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; (long long)i * i <= n; i++)
        if (n % i == 0) return false;
    return true;
}

int main() {
    int T, n;
    cin >> T;
    while (T--) {
        cin >> n;
        cout << (isPrime(n) ? "Lucky" : "Not Lucky") << endl;
    }
    return 0;
}
```

### 关键代码解析

```cpp
bool isPrime(int n) {
    if (n < 2) return false;          // 1 和负数不是质数
    for (int i = 2; (long long)i * i <= n; i++)  // 只需检查到 √n
        if (n % i == 0) return false; // 有因数则非质数
    return true;
}
// 时间复杂度 O(√n)，对 n≤10^6 完全够用
```
