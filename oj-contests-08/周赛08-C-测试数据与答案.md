## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

struct Rider {
    int id, orders, rating;
};

bool cmp(Rider a, Rider b) {
    if (a.rating != b.rating) return a.rating > b.rating;
    return a.orders > b.orders;
}

int main() {
    int n;
    cin >> n;

    Rider riders[100];
    for (int i = 0; i < n; i++)
        cin >> riders[i].id >> riders[i].orders >> riders[i].rating;

    // 筛选达标骑手
    Rider qualified[100];
    int cnt = 0;
    for (int i = 0; i < n; i++)
        if (riders[i].orders >= 20 && riders[i].rating >= 90)
            qualified[cnt++] = riders[i];

    if (cnt == 0) {
        cout << "None" << endl;
    } else {
        sort(qualified, qualified + cnt, cmp);
        for (int i = 0; i < cnt; i++)
            cout << qualified[i].id << endl;
    }

    return 0;
}
```
