## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

struct Student {
    string id, name;
    int chinese, math, total;
};

Student s[105];

bool cmp(const Student& a, const Student& b) {
    if (a.total != b.total) return a.total > b.total;
    if (a.chinese != b.chinese) return a.chinese > b.chinese;
    return a.id < b.id;
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> s[i].id >> s[i].name >> s[i].chinese >> s[i].math;
        s[i].total = s[i].chinese + s[i].math;
    }
    sort(s, s + n, cmp);
    for (int i = 0; i < n; i++)
        cout << s[i].id << " " << s[i].name << " " << s[i].total << endl;
    return 0;
}
```
