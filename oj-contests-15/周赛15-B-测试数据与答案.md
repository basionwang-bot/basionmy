## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int cur = 0, ans = 0;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        if (x >= 60) {
            cur++;
            ans = max(ans, cur);
        } else {
            cur = 0;
        }
    }
    cout << ans << "\n";
    return 0;
}
```
