## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

struct Product {
    string name;
    int dur, perf;
};

bool cmp(const Product& a, const Product& b) {
    if (a.perf != b.perf) return a.perf > b.perf;
    if (a.dur  != b.dur)  return a.dur  > b.dur;
    return a.name < b.name;
}

int main() {
    int n;
    cin >> n;
    vector<Product> products;
    for (int i = 0; i < n; i++) {
        Product p;
        cin >> p.name >> p.dur >> p.perf;
        if (p.dur >= 60 && p.perf >= 60)
            products.push_back(p);
    }
    if (products.empty()) {
        cout << "none\n";
    } else {
        sort(products.begin(), products.end(), cmp);
        for (auto& p : products) cout << p.name << "\n";
    }
    return 0;
}
```
