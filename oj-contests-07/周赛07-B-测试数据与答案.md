**输入**
```

```

## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;

    int a[1000];
    for (int i = 0; i < n; i++) cin >> a[i];

    // 初始化第一个窗口
    int sum = 0;
    for (int i = 0; i < k; i++) sum += a[i];

    int maxSum = sum, startDay = 1;

    // 滑动窗口：每次加入右侧新元素，减去左侧旧元素
    for (int i = k; i < n; i++) {
        sum = sum + a[i] - a[i - k];
        if (sum > maxSum) {
            maxSum = sum;
            startDay = i - k + 2; // 1-indexed 起始天
        }
    }

    cout << maxSum << endl;
    cout << startDay << endl;

    return 0;
}
```

### 关键代码解析

```cpp
// 初始窗口 [0, k-1]（0-indexed），对应第 1 天到第 k 天
int sum = 0;
for (int i = 0; i < k; i++) sum += a[i];

// 滑动：窗口右移一格
// 新窗口 [i-k+1, i]（0-indexed），对应第 (i-k+2) 天到第 (i+1) 天
for (int i = k; i < n; i++) {
    sum = sum + a[i] - a[i - k]; // 加入 a[i]，去掉 a[i-k]
    if (sum > maxSum) {
        maxSum = sum;
        startDay = i - k + 2;   // 起始天（1-indexed）
    }
}
```
