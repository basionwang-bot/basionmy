## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
struct Team { string name, goods; int revenue; };
bool cmp(const Team& a, const Team& b) {
    if (a.revenue != b.revenue) return a.revenue > b.revenue;
    return a.name < b.name;
}
int main() {
    int n; cin >> n;
    vector<Team> res;
    for (int i = 0; i < n; i++) {
        Team t; cin >> t.name >> t.goods >> t.revenue;
        if (t.revenue >= 500) res.push_back(t);
    }
    if (res.empty()) cout << "none\n";
    else { sort(res.begin(), res.end(), cmp); for (auto& t : res) cout << t.name << "\n"; }
    return 0;
}
```
