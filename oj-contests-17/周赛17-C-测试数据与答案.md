## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

struct Case {
    int id;
    string name, level;
};

bool cmp(const Case& a, const Case& b) {
    return a.id < b.id;
}

int main() {
    int n;
    cin >> n;
    vector<Case> cases;
    for (int i = 0; i < n; i++) {
        Case c;
        cin >> c.id >> c.name >> c.level;
        if (c.level == "A") cases.push_back(c);
    }
    if (cases.empty()) {
        cout << "none\n";
    } else {
        sort(cases.begin(), cases.end(), cmp);
        for (auto& c : cases) cout << c.name << "\n";
    }
    return 0;
}
```
