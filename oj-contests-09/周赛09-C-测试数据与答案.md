## 测试点 5（达标线为0，全部输出）

**输入**
```

5 0

7 100
```
**预期输出**
```

```

**说明**：达标线为0，所有人达标；按成绩降序100>50>0。

## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

struct Athlete {
    int id, score;
};

bool cmp(Athlete a, Athlete b) {
    if (a.score != b.score) return a.score > b.score;
    return a.id < b.id;
}

int main() {
    int n, m;
    cin >> n >> m;

    Athlete a[100];
    for (int i = 0; i < n; i++)
        cin >> a[i].id >> a[i].score;

    Athlete qualified[100];
    int cnt = 0;
    for (int i = 0; i < n; i++)
        if (a[i].score >= m)
            qualified[cnt++] = a[i];

    if (cnt == 0) {
        cout << "None" << endl;
    } else {
        sort(qualified, qualified + cnt, cmp);
        for (int i = 0; i < cnt; i++)
            cout << qualified[i].id << " " << qualified[i].score << endl;
    }

    return 0;
}
```
