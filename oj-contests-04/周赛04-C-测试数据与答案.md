## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s;
    cin >> s;
    int l = 0, r = (int)s.size() - 1;
    bool ok = true;
    while (l < r) {
        if (s[l++] != s[r--]) {
            ok = false;
            break;
        }
    }
    cout << (ok ? "YES" : "NO") << endl;
    return 0;
}
```
