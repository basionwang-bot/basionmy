## 测试点 3（总分与短跑都相同，比编号）

**输入**
```
3
C03 A 20 20
C01 B 20 20
C02 C 20 20
```

**预期输出**
```
C01 B 40
C02 C 40
C03 A 40
```

## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

struct Athlete {
    string id, name;
    int run, jump, total;
} a[105];

bool cmp(const Athlete& x, const Athlete& y) {
    if (x.total != y.total) return x.total > y.total;
    if (x.run != y.run) return x.run > y.run;
    return x.id < y.id;
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a[i].id >> a[i].name >> a[i].run >> a[i].jump;
        a[i].total = a[i].run + a[i].jump;
    }
    sort(a, a + n, cmp);
    for (int i = 0; i < n; i++)
        cout << a[i].id << " " << a[i].name << " " << a[i].total << endl;
    return 0;
}
```
