## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

const int N = 1005;
int a[N];

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];
    long long s = 0;
    for (int i = 0; i < n; i++) {
        s += a[i];
        cout << s << endl;
    }
    return 0;
}
```
