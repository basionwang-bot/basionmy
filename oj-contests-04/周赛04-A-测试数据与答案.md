## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

int main() {
    int T;
    cin >> T;
    while (T--) {
        int s;
        cin >> s;
        if (s >= 90) cout << "A\n";
        else if (s >= 80) cout << "B\n";
        else if (s >= 70) cout << "C\n";
        else if (s >= 60) cout << "D\n";
        else cout << "F\n";
    }
    return 0;
}
```
