## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

void update(int& x, int v) {
    x = v;
}

int main() {
    int n, m;
    cin >> n >> m;
    int a[105];
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 0; i < m; i++) {
        int x, v;
        cin >> x >> v;
        update(a[x], v);
    }
    for (int i = 1; i <= n; i++) {
        if (i > 1) cout << " ";
        cout << a[i];
    }
    cout << "\n";
    return 0;
}
```

**关键点**：`void update(int& x, int v)` 中 `x` 是引用，函数内对 `x` 赋值会直接改变调用者传入的变量，无需返回值。
