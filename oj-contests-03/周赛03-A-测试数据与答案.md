## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

int main() {
    int T;
    if (!(cin >> T)) return 0;
    while (T--) {
        int h1, m1, h2, m2;
        cin >> h1 >> m1 >> h2 >> m2;
        int cur = h1 * 60 + m1;
        int alarm = h2 * 60 + m2;
        int diff = alarm - cur;
        if (diff < 0) diff += 24 * 60;
        cout << diff << endl;
    }
    return 0;
}
```
