## 答案（C++ 参考代码）

```cpp
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

struct Course {
    string id, name;
    int credit;
    int selected;
} c[105];

bool cmp(const Course &a, const Course &b) {
    if (a.credit != b.credit) return a.credit > b.credit;       // 学分高的在前
    if (a.selected != b.selected) return a.selected > b.selected; // 已选人数多的在前
    return a.id < b.id;                                         // 课程编号字典序小的在前
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> c[i].id >> c[i].name >> c[i].credit >> c[i].selected;
    }
    sort(c, c + n, cmp);
    for (int i = 0; i < n; i++) {
        cout << c[i].id << " " << c[i].name << " " << c[i].credit << " " << c[i].selected << endl;
    }
    return 0;
}
```
