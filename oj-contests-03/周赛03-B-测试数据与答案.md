## 测试点 3（全部有库存）

**输入**
```
3
5 1 2
```

**预期输出**
```
3
1 2 3
8
```

---

## 答案（C++ 参考代码）

```cpp
#include <iostream>
using namespace std;

const int N = 1005;
int a[N], idx[N];

int main() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> a[i];
    int cnt = 0;
    long long sum = 0;
    for (int i = 1; i <= n; i++) {
        sum += a[i];
        if (a[i] > 0) idx[cnt++] = i;
    }
    cout << cnt << endl;
    if (cnt == 0) {
        cout << "None" << endl;
    } else {
        for (int i = 0; i < cnt; i++) {
            if (i) cout << ' ';
            cout << idx[i];
        }
        cout << endl;
    }
    cout << sum << endl;
    return 0;
}
```
